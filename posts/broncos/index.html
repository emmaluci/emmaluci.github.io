worst50 <- scoring %>% top_n(50, wt= -PF) %>% arrange(desc(PF))

Code 
#bar chart of worst 50 scoring teams over last decade
ggplot() +
  geom_bar(
    data=worst50,
    aes(x=reorder(TeamSeason, PF), weight=PF),
    fill="light grey"
  )+ 
  geom_bar(
    data=worstbroncos,
     aes(x=reorder(TeamSeason, PF), weight=PF),
    fill="#FB4F14"
  )+
coord_flip()+
  labs(
    title="Are the Broncos the worst offensive scoring team?",

    y="Points Scored") +
